CREATE DATABASE shop1
CREATE TABLE `thuonghieu` (
        `MATHUONGHIEU` integer PRIMARY KEY,
        `TENTHUONGHIEU` varchar(100) CHARACTER SET utf8 DEFAULT NULL, 
        `HINHTHUONGHIEU` TEXT,
        `LUOTMUA` integer
        
        );

 CREATE TABLE `shop` (
     `MASHOP`integer PRIMARY KEY AUTO_INCREMENT,
     `TENSHOP` varchar(100) CHARACTER SET utf8 DEFAULT NULL, 
     `HINHSHOP` TEXT,
     `MASP` integer DEFAULT 0
);
         
     CREATE TABLE `loaisanpham`(
         `MALOAISP` integer AUTO_INCREMENT PRIMARY KEY,
         `TENLOAISP` varchar(50) CHARACTER SET utf8,
         `MALOAICHA` integer,
         `HINHLOAISP` TEXT
       
         );
         
	

    CREATE TABLE `sanpham` (
        `MASP`  integer AUTO_INCREMENT PRIMARY KEY,
        `TENSP` varchar(100) CHARACTER SET utf8 DEFAULT NULL, 
        `GIA` decimal(10,0) DEFAULT NULL,
        `ANHLON` TEXT,
        `ANHNHO` TEXT,
        `THONGTIN` TEXT CHARACTER SET  utf8,
        `SOLUONG` int(11) DEFAULT NULL,
        `MALOAISP` integer DEFAULT NULL ,
        `MATHUONGHIEU` integer DEFAULT NULL,
        `MANV` integer,
        `LUOTMUA` integer,
        `LUOTTHICH` integer DEFAULT 0,
		`MASHOP` integer DEFAULT 1
        );
        
     CREATE TABLE `chitietsanpham` (
         `MACHITIET` integer,
         `MASP` integer,
         `TENCHITIET` varchar(100) CHARACTER SET utf8,
         `GIATRI` varchar(50) CHARACTER SET utf8,
         CONSTRAINT `PK_CHITETSP` PRIMARY KEY(`MACHITIET`,`MASP`)
         );
         
      CREATE TABLE `loainhanvien`(
          `MALOAINV` integer PRIMARY KEY,
          `TENLOAINV` varchar(50) CHARACTER SET utf8
          );
     
     CREATE TABLE `nhanvien` (
        `MANV` integer AUTO_INCREMENT PRIMARY KEY,
        `TENNV` varchar(50) CHARACTER SET utf8,
        `TENDANGNHAP` varchar(50),
         `MATKHAU` varchar(30),
         `DIACHI` varchar(50) CHARACTER SET utf8,
         `NGAYSINH` varchar(50),
         `SODT` varchar(12),
         `GIOITINH` varchar(5),
         `MALOAINV` integer
         );
         
     CREATE TABLE `hoadon` (
         `MAHD` integer AUTO_INCREMENT PRIMARY KEY,
         `NGAYMUA` varchar(50),
         `NGAYGIAO` varchar(50),
         `TRANGTHAI` varchar(50) CHARACTER SET utf8 DEFAULT 'chờ kiểm duyệt',
         `TENNGUOINHAN` varchar(50) CHARACTER SET utf8,
         `SODT` varchar(12),
         `DIACHI` varchar(50) CHARACTER SET utf8,
         `HINHTHUCTHANHTOAN` varchar(20) CHARACTER SET utf8
         
         );
         
       CREATE TABLE `chitiethoadon`(
           `MAHD` integer,
           `MASP` integer,
           `SOLUONG` integer,
           CONSTRAINT `PK_CTHD` PRIMARY KEY(`MAHD`,`MASP`)
           );
           
        CREATE TABLE `danhgia`(
         `MADG` varchar(100) NOT NULL,
         `MASP` integer,
	 `TENDANGNHAP` varchar(50) CHARACTER SET utf8,
	 `HINHDANGNHAP` TEXT,
	 `SOSAO` integer,			
        `NOIDUNG` varchar(50) CHARACTER SET utf8,
	 `TENTHIETBI` varchar(150),
         `NGAYDANHGIA` varchar(20),
	    CONSTRAINT PR_DG PRIMARY KEY(MADG,MASP),
            CONSTRAINT FORR_MASP  FOREIGN KEY(`MASP`) REFERENCES `sanpham`(`MASP`)
            
            );

	CREATE TABLE `khuyenmai` (
    `MAKM` integer AUTO_INCREMENT PRIMARY KEY,
    `MALOAISP` integer,
    `TENKM` varchar(100) CHARACTER SET utf8,
    `NGAYBATDAU` varchar(50),
    `NGAYKETTHUC` varchar(50),
    `HINHKHUYENMAI` TEXT,
    CONSTRAINT FR_KM FOREIGN KEY(`MALOAISP`) REFERENCES `loaisanpham`(`MALOAISP`)
    );
 
 CREATE TABLE `chitietkhuyenmai`(
     `MAKM` integer,
     `MASP` integer,
     `PHANTRAMKM` integer,
     CONSTRAINT PK_CTTK PRIMARY KEY (`MAKM`,`MASP`),
     CONSTRAINT FR_CTTK FOREIGN KEY (`MAKM`) REFERENCES `khuyenmai`(`MAKM`)
     );
     
     
  CREATE TABLE `binhluan` (
      `MABL` integer AUTO_INCREMENT PRIMARY KEY,
      `TIEUDE` varchar(1000) CHARACTER SET utf8,
      `MANV`  integer,
      CONSTRAINT FR_BL FOREIGN KEY (`MANV`) REFERENCES `nhanvien`(`MANV`)
      );
      
   CREATE TABLE `chitietbinhluan` (
       `MABL` integer,
       `MANV` integer,
       `NOIDUNGBL` TEXT,
       `NGAYBL` TEXT,
       	CONSTRAINT PR_CTBL PRIMARY KEY(`MABL`,`MANV`),
        CONSTRAINT FR_CTBL FOREIGN KEY (`MANV`) REFERENCES `nhanvien`(`MANV`)
       );
            
               
       

        ALTER TABLE `sanpham` 
        ADD CONSTRAINT `FK_SP` FOREIGN KEY (`MATHUONGHIEU`) REFERENCES `thuonghieu`(`MATHUONGHIEU`),
        ADD CONSTRAINT `FK_SP_LOAISP1` FOREIGN KEY (`MALOAISP`) REFERENCES `loaisanpham`(`MALOAISP`);
        
        ALTER TABLE `chitietsanpham` 
        ADD CONSTRAINT `FK_CTSP_SP` FOREIGN KEY (`MASP`) REFERENCES `SANPHAM`(`MASP`)
        ;
        
        ALTER TABLE `nhanvien` 
        ADD CONSTRAINT `FK_NV_MLNV` FOREIGN KEY (`MALOAINV`) REFERENCES `loainhanvien`(`MALOAINV`)
        ;
        
        
        ALTER TABLE `chitiethoadon` 
        ADD CONSTRAINT `FK_CTHD_MAHD` FOREIGN KEY (`MAHD`) REFERENCES `hoadon`(`MAHD`),
        ADD CONSTRAINT `FK_CTHD_SP` FOREIGN KEY (`MASP`) REFERENCES `sanpham`(`MASP`)
        ;
        
        ALTER TABLE `danhgia` 
        ADD CONSTRAINT `FK_DG_MASP` FOREIGN KEY (`MASP`) REFERENCES `sanpham`(`MASP`)  ;
